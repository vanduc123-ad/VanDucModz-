local function new(class, props)
    local obj = Instance.new(class)
    if props then
        for k,v in pairs(props) do
            if k == "Parent" then obj.Parent = v else pcall(function() obj[k] = v end) end
        end
    end
    return obj
end

local Tabs = {}
local function createTab(name)
    local btn = new("TextButton", {Parent = tabList, Text = name, Size = UDim2.new(1,-8,0,CONF.TAB_BUTTON_HEIGHT),
        BackgroundColor3 = CONF.BACK, TextColor3 = CONF.TEXT, Font = CONF.FONT, TextSize = 18, AutoButtonColor = true})
    new("UICorner", {Parent = btn, CornerRadius = UDim.new(0,8)})
    local frame = new("Frame", {Parent = content, Size = UDim2.new(1,0,1,0), BackgroundTransparency = 1, Visible = false})
    Tabs[name] = frame
    btn.MouseButton1Click:Connect(function()
        for t,f in pairs(Tabs) do f.Visible = false end
        frame.Visible = true
    end)
    return frame
end
